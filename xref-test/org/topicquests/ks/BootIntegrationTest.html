162 	@Test
163 	public void testPropertyTypeSubclasses() throws IOException, ParseException {
164 		IResult result = database.listSubclassNodes("PropertyType", 0, 100, null, null, credentials);
165 		assertEquals("", result.getErrorString());
166 
167 		List<ISubjectProxy> subjectProxyList = ((List<ISubjectProxy>) result.getResultObject())
168 				.parallelStream()
169 				.sorted(Comparator.comparing(subjectProxy -> subjectProxy.getData().get("lox").toString()))
170 				.collect(Collectors.toList());
171 		assertNotNull(subjectProxyList);
172 
173 		String coreString = Files.lines(Paths.get("data/bootstrap/core.json"), StandardCharsets.UTF_8).collect(Collectors.joining("\n"));
174 		String learnerString = Files.lines(Paths.get("data/bootstrap/learner.json"), StandardCharsets.UTF_8).collect(Collectors.joining("\n"));
175 		JSONObject coreObject = (JSONObject) jsonParser.parse(coreString);
176 		JSONObject learnerObject = (JSONObject) jsonParser.parse(learnerString);
177 		JSONArray coreArray = (JSONArray) coreObject.get("nodes");
178 		JSONArray learnerArray = (JSONArray) learnerObject.get("nodes");
179 		coreArray.merge(learnerArray);
180 		List<String> expectedList = coreArray
181 				.parallelStream()
182 				.map(JSONObject.class::cast)
183 				.filter(json -> "PropertyType".equals(json.getAsString("sbOf")))
184 				.map(json -> {
185 					json.put("crtr", "SystemUser");
186 					json.put("_ver", "%s");
187 					json.put("lEdDt", "%s");
188 					json.put("lEdDt", "%s");
189 					json.put("isFdrtd", false);
190 					json.put("trCl", new String[]{"TypeType", "PropertyType"});
191 					json.put("crDt", "%s");
192 					json.put("isPrv", false);
193 					return json;
194 				})
195 				.collect(Collectors.toMap(
196 						json -> json.getAsString("lox"),
197 						json -> json,
198 						(first, second) -> first
199 				))
200 				.values()
201 				.parallelStream()
202 				.sorted(Comparator.comparing(json -> json.getAsString("lox")))
203 				.map(json -> json.toJSONString())
204 				.collect(Collectors.toList());
205 		assertEquals(expectedList.size(), subjectProxyList.size());
206 
207 		String[] expectedJSONArray = IntStream.range(0, expectedList.size())
208 				.mapToObj(i -> String.format(
209 						expectedList.get(i),
210 						subjectProxyList.get(i).getData().get("_ver"),
211 						subjectProxyList.get(i).getData().get("lEdDt"),
212 						subjectProxyList.get(i).getData().get("crDt")
213 				))
214 				.toArray(String[]::new);
215 		String[] subjectProxyJSONArray = subjectProxyList
216 				.parallelStream()
217 				.map(subjectProxy -> subjectProxy.toJSONString())
218 				.toArray(String[]::new);
219 		assertArrayEquals(expectedJSONArray, subjectProxyJSONArray);
220 		assertEquals(
221 				String.format(
222 						"{\"crtr\":\"SystemUser\",\"_ver\":\"%s\",\"lEdDt\":\"%s\",\"label\":\"BacklinkListProperty\",\"isFdrtd\":false,\"trCl\":[\"TypeType\",\"PropertyType\"],\"crDt\":\"%s\",\"sbOf\":\"PropertyType\",\"lox\":\"BacklinkListProperty\",\"sIco\":\"\\/images\\/snowflake_sm.png\",\"isPrv\":false,\"details\":\"Topic Map upper BacklinkList property type\",\"lIco\":\"\\/images\\/snowflake.png\"}",
223 						subjectProxyList.get(0).getData().get("_ver"),
224 						subjectProxyList.get(0).getData().get("lEdDt"),
