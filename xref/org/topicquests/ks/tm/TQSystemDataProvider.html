182 	 * @see org.topicquests.ks.api.ITQDataProvider#getNode(java.lang.String, org.topicquests.ks.api.ITicket)
183 	 */
184 	@Override
185 	public IResult getNode(String locator, ITicket credentials) {
186 		Object nx = nodeCache.get(locator);
187 		IResult result = null;
188 		if (nx != null) {
189 			result = new ResultPojo();
190 			result.setResultObject(nx);
191 		} else {
192 			result = database.getNodeAsJSONObject(locator, _INDEX);
193 			JSONObject jo = (JSONObject)result.getResultObject();
194 			System.out.println("GETNODE "+locator+" "+jo);
195 			result.setResultObject(null); //default in case bad credentials
196 			if (jo != null) {
197 
198 				ISubjectProxy n = new SubjectProxy(jo);
199 				int what = this.checkCredentials(n, credentials);
200 				if (what == 1) {
201 					result.setResultObject(n);
202 					if (!n.getIsPrivate())
203 						nodeCache.add(locator, n);
204 				} else if (what == 0){
205 					result.addErrorString(IErrorMessages.INSUFFICIENT_CREDENTIALS);
206 				} else {
207 					result.addErrorString(IErrorMessages.NODE_REMOVED);
208 				}
209 			}
210 		}
211 		return result;
212 	}

284 	 */
285 	@Override
286 	public IResult getVirtualNodeIfExists(ISubjectProxy node,
287 			ITicket credentials) {
288 		IResult result = new ResultPojo();
289 		List<String> tups = node.listMergeTupleLocators();
290 		//a merged node should have just one-- that is tups.size() == 1
291 		//a virtualnode could have many
292 		if (node.getIsVirtualProxy() ) {
293 			result.setResultObject(node);
294 		} else if (tups != null) {
295 			String tuplox = tups.get(0);
296 			IResult r = this.getNode(tuplox, credentials);
297 			ITuple tup = (ITuple)r.getResultObject();
298 			if (r.hasError()) result.addErrorString(r.getErrorString());
299 			//this ITuple has this node as object, virtualnode as subject
300 			//see BaseVirtualizer.relateNodes
301 			tuplox = tup.getSubjectLocator();
302 			r = this.getNode(tuplox, credentials);
303 			result.setResultObject(r.getResultObject());
304 			if (r.hasError()) result.addErrorString(r.getErrorString());
305 		}
306 		return result;
307 	}

879 	private void recursiveWalkDownTree(IResult result, ITreeNode root, 
880 			int maxDepth, int curDepth, int start, int count, String sortBy, String sortDir, ITicket credentials) {
881 		System.out.println("RECURSIVEWALK "+root.getNodeLocator()+" "+loopStopper);
882 		//stopping rule
883 		if (curDepth == 0)
884 			return;
885 		//Given this root, grab its children, then recurse on them
886 		String lox = root.getNodeLocator();
887 		if (loopStopper.contains(lox))
888 			return;
889 		loopStopper.add(lox);
890 		//Note: the day will come when -1 will bite us in the butt due to huge
891 		//collections
892 		IResult r = this.listSubclassNodes(lox, start, count, sortBy, sortDir, credentials);
893 		if (r.hasError())
894 			result.addErrorString(r.getErrorString());
895 		List<ISubjectProxy>kids = null;
896 		Iterator<ISubjectProxy>itr;
897 		ISubjectProxy snapper;
898 		ITreeNode child;
899 		if (r.getResultObject() != null) {
900 			kids = (List<ISubjectProxy>)r.getResultObject();
901 			itr = kids.iterator();
902 			while (itr.hasNext()) {
903 				//get the kid
904 				snapper = itr.next();
905 				child = new TreeNode(snapper.getLocator(), snapper.getLabel("en"));
906 				root.addSubclassChild(child);
907 				//now populate it
908 				recursiveWalkDownTree(result,child,maxDepth, --curDepth, start, count, sortBy, sortDir, credentials);
909 			}
910 		}
911 		r = this.listInstanceNodes(lox, 0, 200, null, null, credentials);
912 		if (r.hasError())
913 			result.addErrorString(r.getErrorString());
914 		if (r.getResultObject() != null) {
915 			kids = (List<ISubjectProxy>)r.getResultObject();
916 			itr = kids.iterator();
917 			while (itr.hasNext()) {
918 				//get the kid
919 				snapper = itr.next();
920 				child = new TreeNode(snapper.getLocator(), snapper.getLabel("en"));
921 				root.addInstanceChild(child);
922 				//now populate it
923 				recursiveWalkDownTree(result,child,maxDepth, --curDepth,start,count,sortBy, sortDir, credentials);
924 			}
