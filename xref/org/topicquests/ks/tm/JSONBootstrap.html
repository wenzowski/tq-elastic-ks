35  /**
36   * @author Park
37   * A class to replace hard-coded bootstrapping with
38   * file loading from JSON files

40  public class JSONBootstrap {
41  	private SystemEnvironment environment;
42  	private ITQDataProvider database;
43  	private String userId = ITQCoreOntology.SYSTEM_USER;
44  	private ITicket credentials;
45  	private final String path = "data/bootstrap/";
46  	private TextFileHandler handler;

48  	/**
49  	 * 
50  	 */
51  	public JSONBootstrap(SystemEnvironment env) {
52  		this.environment = env;
53  		this.database = environment.getDatabase();
54  		this.credentials = new TicketPojo();
55  		credentials.setUserLocator(userId);
56  		handler = new TextFileHandler();
57  		System.out.println("JSONBOOTSTRAP+");
58  	}

60  	public IResult bootstrap() {
61  		System.out.println("JSONBootstrap- ");
62  		environment.logDebug("JSONBootstrap- ");
63  		IResult result = new ResultPojo();
64  		IResult r = database.getNode(ITQCoreOntology.TYPE_TYPE, credentials);
65  		if (r.hasError())
66  			result.addErrorString(r.getErrorString());
67  		environment.logDebug("JSONBootstrap-1 "+r.getErrorString()+" | "+r.getResultObject());
68  		if (r.getResultObject() == null) {
69  			File dir = new File(path);
70  			System.out.println("JSONBOOTSTRAP.bootstrap "+dir.getAbsolutePath());
71  			File files [] = dir.listFiles();
72  			int len = files.length;
73  			File f;
74  			r = null;
75  			for (int i=0;i<len;i++) {
76  				f = files[i];
77  				System.out.println(f.getAbsolutePath());
78  				if (f.getName().endsWith(".json")) {
79  					r = importJSONFile(f);
80  					if (r.hasError())
81  						result.addErrorString(r.getErrorString());
82  				}
83  			}
84  		} else {
85  			ISubjectProxy p = (ISubjectProxy)r.getResultObject();
86  			System.out.println("FOO "+p.toJSONString());
87  		}
88  		return result;
89  	}

91  	/**
92  	 * Testing the first line of each file. Return <code>true</code>
93  	 * if this class already exists.
94  	 * @param p
95  	 * @return
96  	 */
97  	private boolean seenThis(JSONObject p) {
98  		boolean result = false;
99  		String lox = (String)p.get("lox");
100 		IResult r = database.getNode(lox, credentials);
101 		result = (r.getResultObject() != null);
102 		//System.out.println("SEEN THIS "+lox+" "+r.getResultObject());
103 		return result;
104 	}

106 	private IResult importJSONFile(File f) {
107 		environment.logDebug(f.getName());
108 		IResult result = new ResultPojo();
109 		String json = handler.readFile(f);
110 		JSONParser p = new JSONParser(JSONParser.MODE_JSON_SIMPLE);
111 		try {
112 			JSONObject jo = (JSONObject)p.parse(json);
113 			List<JSONObject> o = (List<JSONObject>)jo.get("nodes");
114 			if (o != null) {
115 				IResult r;
116 				Iterator<JSONObject>itr = o.iterator();
117 				JSONObject x;
118 				boolean isFirst = true;
119 				while (itr.hasNext()) {
120 					x = itr.next();
121 					if (x.get("lox") != null) {
122 						if (isFirst) {
123 							if (seenThis(x)) {
124 								return result;
125 							}
126 							isFirst = false;
127 						}
128 						r = buildProxy(x);
129 						if (r.hasError())
130 							result.addErrorString(r.getErrorString());
131 					}
132 				}
133 			} else {
134 				result.addErrorString(f.getName()+" MISSING Data");
135 			}
136 		} catch (Exception e) {
137 			environment.logError("JSONBootstrap1 "+e.getMessage(), e);
138 			result.addErrorString(e.getMessage());
139 		}
140 		return result;
141 	}

143 	private IResult buildProxy(JSONObject jo)  {
144 		environment.logDebug("Bootstrap "+jo.toJSONString());
145 		IResult result = new ResultPojo();
146 		try {
147 			ISubjectProxy n = new SubjectProxy(jo);
148 			n.setCreatorId(userId);
149 			Date d = new Date();
150 			String sub = (String)n.getProperty("sbOf");
151 			if (sub != null)
152 				n.addTransitiveClosureLocator(sub);
153 			n.setDate(d);
154 			n.setLastEditDate(d);
155 			n.setIsFederated(false);
156 			n.setIsPrivate(false);
157 			n.setVersion(Long.toString(System.currentTimeMillis()));
158 			IResult r = database.putNode(n);
159 			if (r.hasError())
160 				result.addErrorString("JSONBootstrap2 "+r.getErrorString());
161 		} catch (Exception e) {
162 			environment.logError("JSONBootstrap3 "+e.getMessage(), e);
163 			result.addErrorString(e.getMessage());
164 		}
165 		return result;
166 	}
