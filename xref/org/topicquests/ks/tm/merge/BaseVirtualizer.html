201 	/**
202 	 * Forge a merge relation--not using the version in INodeModel
203 	 * @param virtualNode
204 	 * @param targetNode
205 	 * @param userId
206 	 * @param smallImagePath
207 	 * @param largeImagePath
208 	 * @param isTransclude
209 	 * @param isPrivate
210 	 * @param mergeData
211 	 * @return locator of the created tuple
212 	 */
213 	private IResult relateNodes(ISubjectProxy virtualNode, ISubjectProxy targetNode,
214 			String userId, String smallImagePath,
215 			String largeImagePath, boolean isTransclude, boolean isPrivate, Map<String, Double> mergeData) {
216 		String relationTypeLocator = ITQCoreOntology.MERGE_ASSERTION_TYPE;
217 		database.removeFromCache(virtualNode.getLocator());
218 		database.removeFromCache(targetNode.getLocator());
219 		IResult result = new ResultPojo();
220 		String signature = virtualNode.getLocator()+ITQCoreOntology.MERGE_ASSERTION_TYPE+targetNode.getLocator();
221 		//NOTE that we make the tuple an instance of the relation type, not of TUPLE_TYPE
222 		ITuple t = (ITuple)nodeModel.newInstanceNode(signature, relationTypeLocator, 
223 				virtualNode.getLocator()+" "+relationTypeLocator+" "+targetNode.getLocator(), "en", userId, smallImagePath, largeImagePath, isPrivate);
224 		t.setIsTransclude(isTransclude);
225 		//NOTE: tuple object is always merged node
226 		t.setObject(targetNode.getLocator());
227 		t.setObjectType(ITQCoreOntology.NODE_TYPE);
228 		//NOTE: tuple subject is always virtualProxy
229 		t.setSubjectLocator(virtualNode.getLocator());
230 		t.setSubjectType(ITQCoreOntology.NODE_TYPE);
231 		Iterator<String>itx = mergeData.keySet().iterator();
232 		String reason;
233 		while (itx.hasNext()) {
234 			reason = itx.next();
235 			t.addMergeReason(reason+" "+mergeData.get(reason));
236 		}
237 		IResult x = database.putNode(t);
238 		environment.logDebug("MergeBean.relateNodes "+virtualNode.getLocator()+" "+targetNode.getLocator()+" "+t.getLocator());
239 		if (x.hasError())
240 			result.addErrorString(x.getErrorString());
241 		String tLoc = t.getLocator();
242 		//save the tuple's locator in the output
243 		result.setResultObject(tLoc);
244 		addPropertyValue(virtualNode,ITQCoreOntology.MERGE_TUPLE_PROPERTY,tLoc);
245 		//add the value to the node after it's been surgically added
246 		virtualNode.addMergeTupleLocator(tLoc);
247 		targetNode.addMergeTupleLocator(tLoc);
248 		changePropertyValue(targetNode,ITQCoreOntology.MERGE_TUPLE_PROPERTY,tLoc);
249 		environment.logDebug("MergeBean.relateNodes+ "+result.getErrorString());
250 		return result;
251 	}

282 	// Rewiring the graph
283 	// CASE: Merged nodes
284 	// 	A Node was merged with another node.
285 	// 	A VirtualNode was created
286 	// 	For every place in the graph where either node is referenced
287 	//		surgically replace that node locator with the virtual node locator
288 	// CASE: Merged tuples
289 	//  A Tuple has merged with another tuple
290 	//		NOTE: these are NOT Merge-related tuples, only knowledge graph tuples
291 	//  For every node which references a merged tuple in its tuple field
292 	//    	surgically replace that tuple locator with the new tuple locator
293 	//        	this entails overwriting a tuple *in place* in a list of tuples
294 	//          and doing an update on that list in the database
295 	/////////////////////////////////////////////////////////////////
296 	// Emergent Issue
297 	//	If a new node is saved and it is immediately related to another node
298 	//		it will be in a merge process while the relation is occuring.
299 	//		THIS means that there MIGHT be NO TUPLE available to be captured
300 	//		during the merge. Thus, the VirtualNode MIGHT end up with no tuple

302 	/**
303 	 * Substitute <code>virtualProxyLocator</code> for all hits of <code>mergedProxyLocator</code>
304 	 * @param mergedProxyLocator
305 	 * @param virtualProxyLocator
306 	 * @return
307 	 */
308 	IResult reWireNodeGraph(String mergedProxyLocator, String virtualProxyLocator, String mergeTupleLocator, ITicket credentials) {
309 		//this really must deal with tuples first
310 		//TODO sort out what other propertyTypes entail symbolic links, then
311 		//chase those
312 		environment.logDebug("MergeBean.reWireNodeGraph- "+mergedProxyLocator+" "+virtualProxyLocator+" "+mergeTupleLocator);
313 		IResult result = new ResultPojo();
314 		//Find all tuples where mergedProxyLocator isA subject and fix them
315 		//TODO PROBABLY NEED to use an iterator here
316 		IResult xx = tupleQuery.listTuplesBySubject(mergedProxyLocator, 0,-1, null, null, credentials);
317 		if (xx.hasError())
318 			result.addErrorString(xx.getErrorString());
319 		List<ISubjectProxy>n = (List<ISubjectProxy>)xx.getResultObject();
320 		ITuple t;
321 		Iterator<ISubjectProxy>itr;
322 		IResult surgR;
323 		if (n != null && !n.isEmpty()) {
324 			//time for surgery
325 			itr = n.iterator();
326 			while (itr.hasNext()) {
327 				t = (ITuple)itr.next();
328 				if (!t.getLocator().equals(mergeTupleLocator)) {
329 					environment.logDebug("MergeBean.reWireGraph-1 "+t.getLocator());
330 					surgR = performTupleSurgery(t,virtualProxyLocator,true);
331 					if (surgR.hasError())
332 						result.addErrorString(surgR.getErrorString());
333 					//notice the distinct possibility that no surgery got performed
334 					//and the topic map will have errors
335 				}
336 			}
337 		}
338 		//Find all tuples where mergedProxyLocator isA object and fix them
339 		//TODO PROBABLY need an iterator here
340 		xx = tupleQuery.listTuplesByObjectLocator(mergedProxyLocator, 0,-1, null, null, credentials);
341 		if (xx.hasError())
342 			result.addErrorString(xx.getErrorString());
343 		n = (List<ISubjectProxy>)xx.getResultObject();
344 		if (n != null && !n.isEmpty()) {
345 			//time for surgery
346 			itr = n.iterator();
347 			while (itr.hasNext()) {
348 				t = (ITuple)itr.next();
349 				if (!t.getLocator().equals(mergeTupleLocator)) {
350 					environment.logDebug("MergeBean.reWireGraph-2 "+t.getLocator());
351 					surgR = performTupleSurgery(t,virtualProxyLocator,false);
352 					if (surgR.hasError())
353 						result.addErrorString(surgR.getErrorString());
354 					//notice the distinct possibility that no surgery got performed
355 					//and the topic map will have errors
356 				}
357 			}
358 		}
359 		return result;
360 	}
