401 	 * @param parentLocator
402 	 * @return does not return <code>null</code>
403 	 */
404 	private List<String> listTransitiveClosure(String parentLocator) {
405 		IResult x = database.getNode(parentLocator, credentials);
406 		ISubjectProxy n = (ISubjectProxy)x.getResultObject();
407 		List<String>result = new ArrayList<String>();
408 		if (n != null) {
409 			//clone the list
410 			List<String> temp  = n.listTransitiveClosure();
411 			if (temp != null) {
412 				Iterator<String>itr = temp.iterator();
413 				while (itr.hasNext())
414 					result.add(itr.next());
415 			}
416 		}
417 		return result;
418 	}
