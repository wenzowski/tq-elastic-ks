18  /**
19   * @author park
20       * Interface for marker comparators.  During graph matching, and in other situations,
21       * the Notio package may need to establish whether two markers are equivalent (refer
22       * to the same entity).  
23    	 * This decision is likely to be application-specific since markers may refer to 
24    	 * entities outside of the scope of the Notio layer, or the application may wish to 
25    	 * form relationships when comparisons are made.  In order to allow applications to
26    	 * specify the exact rules, they may provide an implementation of this interface as
27    	 * part of a MatchingScheme.  The comparator will be called whenever two markers are
28    	 * compared.  Possible uses include:
29    	 * <UL>
30    	 *   <LI>comparing complex data objects such as images referred to by markers
31    	 *   <LI>testing and creating equivalencies between markers due to assertions about them
32    	 *   <LI>references to external database for which the markers are keys
33    	 * </UL>
34   *
35   */
36  public interface IMarkerComparator {
37  
38  	boolean compareMarkers(IMarker firstMarker, IMarker secondMarker);
39  }
