154 		IResult result = new ResultPojo();
155 		System.out.println("PopulateVertex "+v.toJSONString());
156 		List<String> edges = null;
157 		Iterator<IEdge>ite = v.listEdges(IGraphOntology.DIRECTION_IN, (String [])null);
158 		if (ite == null) { // alredy populated?
159 			edges = (List<String>)v.getData().get(IGraphOntology.IN_EDGE_ID_LIST_PROPERTY_TYPE);
160 			if (edges != null)
161 				grabEdges(result, IGraphOntology.DIRECTION_IN, v, edges);
162 		}
163 		ite = v.listEdges(IGraphOntology.DIRECTION_OUT, (String [])null);
164 		if (ite == null) { // already populated?
165 			edges = (List<String>)v.getData().get(IGraphOntology.OUT_EDGE_ID_LIST_PROPERTY_TYPE);
166 			if (edges != null)
167 				grabEdges(result, IGraphOntology.DIRECTION_OUT, v, edges);
168 		}
169 		result.setResultObject(v);
170 		//return it to the cache
171 		vertexCache.add(v.getId(), v);
172 		return result;

183 		String eid;
184 		Iterator<String>itr = edges.iterator();
185 		IResult rx;
